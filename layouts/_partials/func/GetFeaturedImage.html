{{/* GetFeaturedImage This partial gets the url for featured image for a given page. If a featured_image was set in the page's front matter, then that will be used. If not set, this will search page resources to find an image that contains the word "cover", and if found, returns the path to that resource. If no featured_image was set, and there's no "cover" image in page resources, then this partial returns an empty string (which evaluates to false). Uses relative paths for static images so thumbnails load correctly regardless of baseURL (e.g. maepopi.github.io vs custom domain). @return RelPermalink to featured image, or an empty string if not found. */}} {{/* Declare a new string variable, $linkToCover */}} {{ $linkToCover := "" }} {{ $matches := "feature,cover" }} {{/* Use the value from front matter if present */}} {{ with .Params.featured_image }} {{/* If we find a Page Resource matching the exact value, we use it instead. */}} {{ with $.Resources.GetMatch . }} {{ $linkToCover = .RelPermalink }} {{ else }} {{/* Static image: use relative path so it works on any domain (GitHub Pages, custom domain) */}} {{ $path := strings.Trim . "/" }} {{ if hasPrefix $path "http" }} {{ $linkToCover = $path }} {{ else }} {{ $linkToCover = printf "/%s" $path }} {{ end }} {{ end }} {{/* Find the first image with 'cover' in the name in this page bundle. */}} {{ else }} {{ with .Resources.ByType "image" }} {{ with .GetMatch (fmt.Printf "**{%s}*" $matches) }} {{ $linkToCover = .RelPermalink }} {{ end }} {{ end }} {{ end }} {{/* return either a permalink, or an empty string. Note that partials can only have a single return statement, so it needs to be at the end of the partial (and not in the if block) */}} {{ return $linkToCover }}
